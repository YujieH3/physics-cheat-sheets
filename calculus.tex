\begin{multicols}{2}
    Tangent half-angle substitution $t = \tan \frac{x}{2}$
    \begin{gather*}
        \dd x = \frac{2}{1 + t^2}\dd t,\quad \sin{x} = \frac{2t}{1 + t^2},\quad \cos{x} = \frac{1 - t^2}{1 + t^2}
    \end{gather*}
